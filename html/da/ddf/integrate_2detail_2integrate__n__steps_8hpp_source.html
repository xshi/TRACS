96 template< class Stepper , class System , class State , class Time , class Observer>
97 Time integrate_n_steps(
98  Stepper stepper , System system , State &start_state ,
99  Time start_time , Time dt , size_t num_of_steps ,
100  Observer observer , dense_output_stepper_tag )
101 {
102  typename odeint::unwrap_reference< Observer >::type &obs = observer;
103  typename odeint::unwrap_reference< Stepper >::type &st = stepper;
104 
105  Time time = start_time;
106  const Time end_time = start_time + static_cast< typename unit_value_type<Time>::type >(num_of_steps) * dt;
107 
108  st.initialize( start_state , time , dt );
109 
110  size_t step = 0;
111 
112  while( step < num_of_steps )
113  {
114  while( less_with_sign( time , st.current_time() , st.current_time_step() ) )
115  {
116  st.calc_state( time , start_state );
117  obs( start_state , time );
118  ++step;
119  // direct computation of the time avoids error propagation happening when using time += dt
120  // we need clumsy type analysis to get boost units working here
121  time = start_time + static_cast< typename unit_value_type<Time>::type >(step) * dt;
122  }
123 
124  // we have not reached the end, do another real step
125  if( less_with_sign( static_cast<Time>(st.current_time()+st.current_time_step()) ,
126  end_time ,
127  st.current_time_step() ) )
128  {
129  st.do_step( system );
130  }
131  else if( less_with_sign( st.current_time() , end_time , st.current_time_step() ) )
132  { // do the last step ending exactly on the end point
133  st.initialize( st.current_state() , st.current_time() , static_cast<Time>(end_time - st.current_time()) );
134  st.do_step( system );
135  }
136  }
137 
138  while( st.current_time() < end_time )
139  {
140  if( less_with_sign( end_time ,
141  static_cast<Time>(st.current_time()+st.current_time_step()) ,
142  st.current_time_step() ) )
143  st.initialize( st.current_state() , st.current_time() , static_cast<Time>(end_time - st.current_time()) );
144  st.do_step( system );
145  }
146 
147  // observation at end point, only if we ended exactly on the end-point (or above due to finite precision)
148  obs( st.current_state() , end_time );

256  {
257  BOOST_USING_STD_MAX();
258  using std::abs;
259  typedef typename norm_result_type< boost::array< T , dim > >::type result_type;
260  result_type init = static_cast< result_type >( 0 );
261  for( size_t i=0 ; i<dim ; ++i )
